CREATE TABLE Employee (
    id INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name NVARCHAR(300),
    salary INTEGER,
    department NVARCHAR(300),
    city NVARCHAR(300));
